SQL Requirements
CREATE TABLE issues (
    UID int NOT NULL,
    Name varchar(255) NOT NULL,
    Date varchar(255) NOT NULL,
    Project varchar(255) NOT NULL,
    Details varchar(255),
    Status varchar(255) NOT NULL,
    PRIMARY KEY (UID)
);
